    Cómo usar la API

    Obtener todos los estados

    fetch('src/data/states.json')
  .then(response => response.json())
  .then(states => console.log(states));

    Obtener todas las ciudades

    fetch('src/data/cities.json')
  .then(response => response.json())
  .then(cities => console.log(cities));

    Filtrar ciudades por estado

    fetch('src/data/cities.json')
  .then(response => response.json())
  .then(cities => {
    const stateId = 1; // ID del estado (ej: 1 para Amazonas)
    const citiesInState = cities.filter(city => city.stateId === stateId);
    console.log(citiesInState);
  });
